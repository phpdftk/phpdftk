22/**
23 * Typed accessor surface over a per-element {@see CascadedValues} bag
24 * per `docs/plans/contracts.md` §1D.5. Each getter narrows the
25 * underlying `Value` to the type(s) the property's grammar permits;
26 * properties not explicitly set fall back to the registry's initial
27 * value (via `CascadedValues::get()`).
28 *
29 * The class is intentionally a thin wrapper — it adds no caching or
30 * computation of its own. The cascade resolves percentages, inherits
31 * values, and produces the initial-value fallbacks; ComputedStyle
32 * just exposes them with PHP-static-analysable signatures so
33 * downstream consumers (the box generator, painter, SVG renderer)
34 * don't have to type-check every `?Value` themselves.
35 *
36 * **Type narrowing fallback.** When a property's parsed value doesn't
37 * match its declared type (an invalid `font-weight: foo` slipping
38 * through), the getter returns the property's initial value instead
39 * of throwing — keeping the accessor surface infallible at the cost
40 * of swallowing parser errors that should have been caught earlier.
41 */

392     * CSS Backgrounds 3 §4.4. Accepts the named keyword forms
393     * (`thin` = 1px, `medium` = 3px, `thick` = 5px) in addition
394     * to bare Length.

397    {
398        $v = $this->values->get($prop);
399        if ($v instanceof Length) {
400            return $v;
401        }

404                'thin' => 1.0,
405                'medium' => 3.0,
406                'thick' => 5.0,
407                default => 0.0,
408            };
409            return new Length($px, \Phpdftk\Css\Value\LengthUnit::Px);
410        }
411        return new Length(3.0, \Phpdftk\Css\Value\LengthUnit::Px);
412    }

784    /**
785     * CSS Color 5 §3 — resolve `color-mix(in srgb, A p%, B q%)` to
786     * the linearly-interpolated sRGB Color. Other spaces (lab, lch,
787     * etc.) are passed through unchanged for now; full per-space
788     * mixing math lands with the color engine (Phase 4E).
789     */
790    private function resolveColorMix(?Value $v): ?Value
791    {
792        if (!$v instanceof ColorMix) {
793            return $v;
794        }
795        // Only sRGB-space mixing is supported here; other spaces
796        // fall through to the renderer's color engine.
797        if ($v->space !== ColorSpace::sRGB && $v->space !== ColorSpace::sRGBLinear) {
798            return $v;
799        }
800        $total = $v->percentage1 + $v->percentage2;
801        if ($total <= 0.0) {
802            return $v;
803        }
804        $t = $v->percentage2 / $total;
805        $mix = static fn(float $a, float $b): float => $a * (1.0 - $t) + $b * $t;
806        $a = $v->color1;
807        $b = $v->color2;
808        $alpha = $mix($a->a, $b->a) * $v->alphaMultiplier;
809        return new Color(
810            $mix($a->r, $b->r),
811            $mix($a->g, $b->g),
812            $mix($a->b, $b->b),
813            max(0.0, min(1.0, $alpha)),
814        );
815    }

817    /**
818     * CSS Color 7 §4 — `contrast-color(<color>)`. Resolves to
819     * black or white based on which has higher contrast against
820     * the base color. Computes WCAG 2.x relative luminance and
821     * picks black for the lighter half, white for the darker.
822     *

824     * a CssFunction the engine can't yet resolve) fall through —
825     * the renderer treats `contrast-color(currentcolor)` as
826     * "currentcolor" rather than throwing.
827     */

830        if (!$v instanceof ContrastColor) {

832        }

842    /**
843     * Compute WCAG 2.x relative luminance of an sRGB Color.
844     * Components linearised per the WCAG formula, then weighted
845     * 0.2126 R + 0.7152 G + 0.0722 B.
846     */

852                : (($v + 0.055) / 1.055) ** 2.4;
853        };

855            + 0.7152 * $linear($c->g)

857    }
